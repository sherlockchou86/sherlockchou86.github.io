36 ~LogStreamConsumerBuffer()

38 // std::streambuf::pbase() gives a pointer to the beginning of the buffered part of the output sequence
39 // std::streambuf::pptr() gives a pointer to the current position of the output sequence
40 // if the pointer to the beginning is not equal to the pointer to the current position,
41 // call putOutput() to log the output to the stream
42 if (pbase() != pptr())
43 {
44 putOutput();

48 // synchronizes the stream buffer and returns 0 on success
49 // synchronizing the stream buffer consists of inserting the buffer contents into the stream,
50 // resetting the buffer and flushing the stream
51 virtual int sync()
52 {
53 putOutput();
54 return 0;
55 }

57 void putOutput()
58 {
59 if (mShouldLog)
60 {
61 // prepend timestamp
62 std::time_t timestamp = std::time(nullptr);
63 tm* tm_local = std::localtime(&timestamp);
64 std::cout << "[";
65 std::cout << std::setw(2) << std::setfill('0') << 1 + tm_local->tm_mon << "/";
66 std::cout << std::setw(2) << std::setfill('0') << tm_local->tm_mday << "/";
67 std::cout << std::setw(4) << std::setfill('0') << 1900 + tm_local->tm_year << "-";
68 std::cout << std::setw(2) << std::setfill('0') << tm_local->tm_hour << ":";
69 std::cout << std::setw(2) << std::setfill('0') << tm_local->tm_min << ":";
70 std::cout << std::setw(2) << std::setfill('0') << tm_local->tm_sec << "] ";
71 // std::stringbuf::str() gets the string contents of the buffer
72 // insert the buffer contents pre-appended by the appropriate prefix into the stream
73 mOutput << mPrefix << str();
74 }
75 // set the buffer to empty
76 str("");
77 // flush the stream
78 mOutput.flush();
79 }
