195// Checks whether a point lies in a cell
196bool Cell::containsPoint(double point[])
197{
198 for(int d = 0; d < dimension; d++) {
199 if(corner[d] - width[d] > point[d]) return false;
200 if(corner[d] + width[d] < point[d]) return false;
201 }
202 return true;
203}

206// Default constructor for SPTree -- build tree, too!
207SPTree::SPTree(unsigned int D, double* inp_data, unsigned int N)
208{
209
210 // Compute mean, width, and height of current map (boundaries of SPTree)
211 int nD = 0;
212 double* mean_Y = (double*) calloc(D, sizeof(double));
213 double* min_Y = (double*) malloc(D * sizeof(double)); for(unsigned int d = 0; d < D; d++) min_Y[d] = DBL_MAX;
214 double* max_Y = (double*) malloc(D * sizeof(double)); for(unsigned int d = 0; d < D; d++) max_Y[d] = -DBL_MAX;
215 for(unsigned int n = 0; n < N; n++) {
216 for(unsigned int d = 0; d < D; d++) {
217 mean_Y[d] += inp_data[n * D + d];
218 if(inp_data[nD + d] < min_Y[d]) min_Y[d] = inp_data[nD + d];
219 if(inp_data[nD + d] > max_Y[d]) max_Y[d] = inp_data[nD + d];
220 }
221 nD += D;
222 }
223 for(int d = 0; d < D; d++) mean_Y[d] /= (double) N;
224
225 // Construct SPTree
226 double* width = (double*) malloc(D * sizeof(double));
227 for(int d = 0; d < D; d++) width[d] = fmax(max_Y[d] - mean_Y[d], mean_Y[d] - min_Y[d]) + 1e-5;
228 init(NULL, D, inp_data, mean_Y, width);
229 fill(N);
230
231 // Clean up memory
232 free(mean_Y);
233 free(max_Y);
234 free(min_Y);
235 free(width);
236}

268// Main initialization function
269void SPTree::init(SPTree* inp_parent, unsigned int D, double* inp_data, double* inp_corner, double* inp_width)
270{
271 parent = inp_parent;
272 dimension = D;
273 no_children = 2;
274 for(unsigned int d = 1; d < D; d++) no_children *= 2;
275 data = inp_data;
276 is_leaf = true;
277 size = 0;
278 cum_size = 0;
279
280 boundary = new Cell(dimension);
281 for(unsigned int d = 0; d < D; d++) boundary->setCorner(d, inp_corner[d]);
282 for(unsigned int d = 0; d < D; d++) boundary->setWidth( d, inp_width[d]);
283
284 children = (SPTree**) malloc(no_children * sizeof(SPTree*));
285 for(unsigned int i = 0; i < no_children; i++) children[i] = NULL;
286
287 center_of_mass = (double*) malloc(D * sizeof(double));
288 for(unsigned int d = 0; d < D; d++) center_of_mass[d] = .0;
289
290 buff = (double*) malloc(D * sizeof(double));
291}

321// Insert a point into the SPTree
322bool SPTree::insert(unsigned int new_index)
323{
324 // Ignore objects which do not belong in this quad tree
325 double* point = data + new_index * dimension;
326 if(!boundary->containsPoint(point))
327 return false;
328
329 // Online update of cumulative size and center-of-mass
330 cum_size++;
331 double mult1 = (double) (cum_size - 1) / (double) cum_size;
332 double mult2 = 1.0 / (double) cum_size;
333 for(unsigned int d = 0; d < dimension; d++) center_of_mass[d] *= mult1;
334 for(unsigned int d = 0; d < dimension; d++) center_of_mass[d] += mult2 * point[d];
335
336 // If there is space in this quad tree and it is a leaf, add the object here
337 if(is_leaf && size < QT_NODE_CAPACITY) {
338 index[size] = new_index;
339 size++;
340 return true;
341 }
342
343 // Don't add duplicates for now (this is not very nice)
344 bool any_duplicate = false;
345 for(unsigned int n = 0; n < size; n++) {
346 bool duplicate = true;
347 for(unsigned int d = 0; d < dimension; d++) {
348 if(point[d] != data[index[n] * dimension + d]) { duplicate = false; break; }
349 }
350 any_duplicate = any_duplicate | duplicate;
351 }
352 if(any_duplicate) return true;
353
354 // Otherwise, we need to subdivide the current cell
355 if(is_leaf) subdivide();
356
357 // Find out where the point can be inserted
358 for(unsigned int i = 0; i < no_children; i++) {
359 if(children[i]->insert(new_index)) return true;
360 }
361
362 // Otherwise, the point cannot be inserted (this should never happen)
363 return false;
364}

367// Create four children which fully divide this cell into four quads of equal area
368void SPTree::subdivide() {
369
370 // Create new children
371 double* new_corner = (double*) malloc(dimension * sizeof(double));
372 double* new_width = (double*) malloc(dimension * sizeof(double));
373 for(unsigned int i = 0; i < no_children; i++) {
374 unsigned int div = 1;
375 for(unsigned int d = 0; d < dimension; d++) {
376 new_width[d] = .5 * boundary->getWidth(d);
377 if((i / div) % 2 == 1) new_corner[d] = boundary->getCorner(d) - .5 * boundary->getWidth(d);
378 else new_corner[d] = boundary->getCorner(d) + .5 * boundary->getWidth(d);
379 div *= 2;
380 }
381 children[i] = new SPTree(this, dimension, data, new_corner, new_width);
382 }
383 free(new_corner);
384 free(new_width);
385
386 // Move existing points to correct children
387 for(unsigned int i = 0; i < size; i++) {
388 bool success = false;
389 for(unsigned int j = 0; j < no_children; j++) {
390 if(!success) success = children[j]->insert(index[i]);
391 }
392 index[i] = -1;
393 }
394
395 // Empty parent node
396 size = 0;
397 is_leaf = false;
398}

408// Checks whether the specified tree is correct
409bool SPTree::isCorrect()
410{
411 for(unsigned int n = 0; n < size; n++) {
412 double* point = data + index[n] * dimension;
413 if(!boundary->containsPoint(point)) return false;
414 }
415 if(!is_leaf) {
416 bool correct = true;
417 for(int i = 0; i < no_children; i++) correct = correct && children[i]->isCorrect();
418 return correct;
419 }
420 else return true;
421}

456// Compute non-edge forces using Barnes-Hut algorithm
457void SPTree::computeNonEdgeForces(unsigned int point_index, double theta, double neg_f[], double* sum_Q)
458{
459
460 // Make sure that we spend no time on empty nodes or self-interactions
461 if(cum_size == 0 || (is_leaf && size == 1 && index[0] == point_index)) return;
462
463 // Compute distance between point and center-of-mass
464 double D = .0;
465 unsigned int ind = point_index * dimension;
466 for(unsigned int d = 0; d < dimension; d++) buff[d] = data[ind + d] - center_of_mass[d];
467 for(unsigned int d = 0; d < dimension; d++) D += buff[d] * buff[d];
468
469 // Check whether we can use this node as a "summary"
470 double max_width = 0.0;
471 double cur_width;
472 for(unsigned int d = 0; d < dimension; d++) {
473 cur_width = boundary->getWidth(d);
474 max_width = (max_width > cur_width) ? max_width : cur_width;
475 }
476 if(is_leaf || max_width / sqrt(D) < theta) {
477
478 // Compute and add t-SNE force between point and current node
479 D = 1.0 / (1.0 + D);
480 double mult = cum_size * D;
481 *sum_Q += mult;
482 mult *= D;
483 for(unsigned int d = 0; d < dimension; d++) neg_f[d] += mult * buff[d];
484 }
485 else {
486
487 // Recursively apply Barnes-Hut to children
488 for(unsigned int i = 0; i < no_children; i++) children[i]->computeNonEdgeForces(point_index, theta, neg_f, sum_Q);
489 }
490}

493// Computes edge forces
494void SPTree::computeEdgeForces(unsigned int* row_P, unsigned int* col_P, double* val_P, int N, double* pos_f)
495{
496
497 // Loop over all edges in the graph
498 unsigned int ind1 = 0;
499 unsigned int ind2 = 0;
500 double D;
501 for(unsigned int n = 0; n < N; n++) {
502 for(unsigned int i = row_P[n]; i < row_P[n + 1]; i++) {
503
504 // Compute pairwise distance and Q-value
505 D = 1.0;
506 ind2 = col_P[i] * dimension;
507 for(unsigned int d = 0; d < dimension; d++) buff[d] = data[ind1 + d] - data[ind2 + d];
508 for(unsigned int d = 0; d < dimension; d++) D += buff[d] * buff[d];
509 D = val_P[i] / D;
510
511 // Sum positive force
512 for(unsigned int d = 0; d < dimension; d++) pos_f[ind1 + d] += D * buff[d];
513 }
514 ind1 += dimension;
515 }
516}

519// Print out tree
520void SPTree::print()
521{
522 if(cum_size == 0) {
523 printf("Empty node\n");
524 return;
525 }
526
527 if(is_leaf) {
528 printf("Leaf node; data = [");
529 for(int i = 0; i < size; i++) {
530 double* point = data + index[i] * dimension;
531 for(int d = 0; d < dimension; d++) printf("%f, ", point[d]);
532 printf(" (index = %d)", index[i]);
533 if(i < size - 1) printf("\n");
534 else printf("]\n");
535 }
536 }
537 else {
538 printf("Intersection node with center-of-mass = [");
539 for(int d = 0; d < dimension; d++) printf("%f, ", center_of_mass[d]);
540 printf("]; children are:\n");
541 for(int i = 0; i < no_children; i++) children[i]->print();
542 }
543}
