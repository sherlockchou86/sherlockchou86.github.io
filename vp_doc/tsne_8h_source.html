109// Perform t-SNE
110void TSNE::run(double* X, int N, int D, double* Y, int no_dims, double perplexity, double theta, int rand_seed,
111 bool skip_random_init, int max_iter, int stop_lying_iter, int mom_switch_iter) {
112
113 // Set random seed
114 if (skip_random_init != true) {
115 if(rand_seed >= 0) {
116 printf("Using random seed: %d\n", rand_seed);
117 srand((unsigned int) rand_seed);
118 } else {
119 printf("Using current time as random seed...\n");
120 srand(time(NULL));
121 }
122 }
123
124 // Determine whether we are using an exact algorithm
125 if(N - 1 < 3 * perplexity) { printf("Perplexity too large for the number of data points!\n"); exit(1); }
126 printf("Using no_dims = %d, perplexity = %f, and theta = %f\n", no_dims, perplexity, theta);
127 bool exact = (theta == .0) ? true : false;
128
129 // Set learning parameters
130 float total_time = .0;
131 clock_t start, end;
132 double momentum = .5, final_momentum = .8;
133 double eta = 200.0;
134
135 // Allocate some memory
136 double* dY = (double*) malloc(N * no_dims * sizeof(double));
137 double* uY = (double*) malloc(N * no_dims * sizeof(double));
138 double* gains = (double*) malloc(N * no_dims * sizeof(double));
139 if(dY == NULL || uY == NULL || gains == NULL) { printf("Memory allocation failed!\n"); exit(1); }
140 for(int i = 0; i < N * no_dims; i++) uY[i] = .0;
141 for(int i = 0; i < N * no_dims; i++) gains[i] = 1.0;
142
143 // Normalize input data (to prevent numerical problems)
144 printf("Computing input similarities...\n");
145 start = clock();
146 zeroMean(X, N, D);
147 double max_X = .0;
148 for(int i = 0; i < N * D; i++) {
149 if(fabs(X[i]) > max_X) max_X = fabs(X[i]);
150 }
151 for(int i = 0; i < N * D; i++) X[i] /= max_X;
152
153 // Compute input similarities for exact t-SNE
154 double* P; unsigned int* row_P; unsigned int* col_P; double* val_P;
155 if(exact) {
156
157 // Compute similarities
158 printf("Exact?");
159 P = (double*) malloc(N * N * sizeof(double));
160 if(P == NULL) { printf("Memory allocation failed!\n"); exit(1); }
161 computeGaussianPerplexity(X, N, D, P, perplexity);
162
163 // Symmetrize input similarities
164 printf("Symmetrizing...\n");
165 int nN = 0;
166 for(int n = 0; n < N; n++) {
167 int mN = (n + 1) * N;
168 for(int m = n + 1; m < N; m++) {
169 P[nN + m] += P[mN + n];
170 P[mN + n] = P[nN + m];
171 mN += N;
172 }
173 nN += N;
174 }
175 double sum_P = .0;
176 for(int i = 0; i < N * N; i++) sum_P += P[i];
177 for(int i = 0; i < N * N; i++) P[i] /= sum_P;
178 }
179
180 // Compute input similarities for approximate t-SNE
181 else {
182
183 // Compute asymmetric pairwise input similarities
184 computeGaussianPerplexity(X, N, D, &row_P, &col_P, &val_P, perplexity, (int) (3 * perplexity));
185
186 // Symmetrize input similarities
187 symmetrizeMatrix(&row_P, &col_P, &val_P, N);
188 double sum_P = .0;
189 for(int i = 0; i < row_P[N]; i++) sum_P += val_P[i];
190 for(int i = 0; i < row_P[N]; i++) val_P[i] /= sum_P;
191 }
192 end = clock();
193
194 // Lie about the P-values
195 if(exact) { for(int i = 0; i < N * N; i++) P[i] *= 12.0; }
196 else { for(int i = 0; i < row_P[N]; i++) val_P[i] *= 12.0; }
197
198 // Initialize solution (randomly)
199 if (skip_random_init != true) {
200 for(int i = 0; i < N * no_dims; i++) Y[i] = randn() * .0001;
201 }
202
203 // Perform main training loop
204 if(exact) printf("Input similarities computed in %4.2f seconds!\nLearning embedding...\n", (float) (end - start) / CLOCKS_PER_SEC);
205 else printf("Input similarities computed in %4.2f seconds (sparsity = %f)!\nLearning embedding...\n", (float) (end - start) / CLOCKS_PER_SEC, (double) row_P[N] / ((double) N * (double) N));
206 start = clock();
207
208 for(int iter = 0; iter < max_iter; iter++) {
209
210 // Compute (approximate) gradient
211 if(exact) computeExactGradient(P, Y, N, no_dims, dY);
212 else computeGradient(row_P, col_P, val_P, Y, N, no_dims, dY, theta);
213
214 // Update gains
215 for(int i = 0; i < N * no_dims; i++) gains[i] = (sign(dY[i]) != sign(uY[i])) ? (gains[i] + .2) : (gains[i] * .8);
216 for(int i = 0; i < N * no_dims; i++) if(gains[i] < .01) gains[i] = .01;
217
218 // Perform gradient update (with momentum and gains)
219 for(int i = 0; i < N * no_dims; i++) uY[i] = momentum * uY[i] - eta * gains[i] * dY[i];
220 for(int i = 0; i < N * no_dims; i++) Y[i] = Y[i] + uY[i];
221
222 // Make solution zero-mean
223 zeroMean(Y, N, no_dims);
224
225 // Stop lying about the P-values after a while, and switch momentum
226 if(iter == stop_lying_iter) {
227 if(exact) { for(int i = 0; i < N * N; i++) P[i] /= 12.0; }
228 else { for(int i = 0; i < row_P[N]; i++) val_P[i] /= 12.0; }
229 }
230 if(iter == mom_switch_iter) momentum = final_momentum;
231
232 // Print out progress
233 if (iter > 0 && (iter % 50 == 0 || iter == max_iter - 1)) {
234 end = clock();
235 double C = .0;
236 if(exact) C = evaluateError(P, Y, N, no_dims);
237 else C = evaluateError(row_P, col_P, val_P, Y, N, no_dims, theta); // doing approximate computation here!
238 if(iter == 0)
239 printf("Iteration %d: error is %f\n", iter + 1, C);
240 else {
241 total_time += (float) (end - start) / CLOCKS_PER_SEC;
242 printf("Iteration %d: error is %f (50 iterations in %4.2f seconds)\n", iter, C, (float) (end - start) / CLOCKS_PER_SEC);
243 }
244 start = clock();
245 }
246 }
247 end = clock(); total_time += (float) (end - start) / CLOCKS_PER_SEC;
248
249 // Clean up memory
250 free(dY);
251 free(uY);
252 free(gains);
253 if(exact) free(P);
254 else {
255 free(row_P); row_P = NULL;
256 free(col_P); col_P = NULL;
257 free(val_P); val_P = NULL;
258 }
259 printf("Fitting performed in %4.2f seconds.\n", total_time);
260}

263// Compute gradient of the t-SNE cost function (using Barnes-Hut algorithm)
264static void computeGradient(unsigned int* inp_row_P, unsigned int* inp_col_P, double* inp_val_P, double* Y, int N, int D, double* dC, double theta)
265{
266
267 // Construct space-partitioning tree on current map
268 SPTree* tree = new SPTree(D, Y, N);
269
270 // Compute all terms required for t-SNE gradient
271 double sum_Q = .0;
272 double* pos_f = (double*) calloc(N * D, sizeof(double));
273 double* neg_f = (double*) calloc(N * D, sizeof(double));
274 if(pos_f == NULL || neg_f == NULL) { printf("Memory allocation failed!\n"); exit(1); }
275 tree->computeEdgeForces(inp_row_P, inp_col_P, inp_val_P, N, pos_f);
276 for(int n = 0; n < N; n++) tree->computeNonEdgeForces(n, theta, neg_f + n * D, &sum_Q);
277
278 // Compute final t-SNE gradient
279 for(int i = 0; i < N * D; i++) {
280 dC[i] = pos_f[i] - (neg_f[i] / sum_Q);
281 }
282 free(pos_f);
283 free(neg_f);
284 delete tree;
285}

287// Compute gradient of the t-SNE cost function (exact)
288static void computeExactGradient(double* P, double* Y, int N, int D, double* dC) {
289
290 // Make sure the current gradient contains zeros
291 for(int i = 0; i < N * D; i++) dC[i] = 0.0;
292
293 // Compute the squared Euclidean distance matrix
294 double* DD = (double*) malloc(N * N * sizeof(double));
295 if(DD == NULL) { printf("Memory allocation failed!\n"); exit(1); }
296 computeSquaredEuclideanDistance(Y, N, D, DD);
297
298 // Compute Q-matrix and normalization sum
299 double* Q = (double*) malloc(N * N * sizeof(double));
300 if(Q == NULL) { printf("Memory allocation failed!\n"); exit(1); }
301 double sum_Q = .0;
302 int nN = 0;
303 for(int n = 0; n < N; n++) {
304 for(int m = 0; m < N; m++) {
305 if(n != m) {
306 Q[nN + m] = 1 / (1 + DD[nN + m]);
307 sum_Q += Q[nN + m];
308 }
309 }
310 nN += N;
311 }
312
313 // Perform the computation of the gradient
314 nN = 0;
315 int nD = 0;
316 for(int n = 0; n < N; n++) {
317 int mD = 0;
318 for(int m = 0; m < N; m++) {
319 if(n != m) {
320 double mult = (P[nN + m] - (Q[nN + m] / sum_Q)) * Q[nN + m];
321 for(int d = 0; d < D; d++) {
322 dC[nD + d] += (Y[nD + d] - Y[mD + d]) * mult;
323 }
324 }
325 mD += D;
326 }
327 nN += N;
328 nD += D;
329 }
330
331 // Free memory
332 free(DD); DD = NULL;
333 free(Q); Q = NULL;
334}

337// Evaluate t-SNE cost function (exactly)
338static double evaluateError(double* P, double* Y, int N, int D) {
339
340 // Compute the squared Euclidean distance matrix
341 double* DD = (double*) malloc(N * N * sizeof(double));
342 double* Q = (double*) malloc(N * N * sizeof(double));
343 if(DD == NULL || Q == NULL) { printf("Memory allocation failed!\n"); exit(1); }
344 computeSquaredEuclideanDistance(Y, N, D, DD);
345
346 // Compute Q-matrix and normalization sum
347 int nN = 0;
348 double sum_Q = DBL_MIN;
349 for(int n = 0; n < N; n++) {
350 for(int m = 0; m < N; m++) {
351 if(n != m) {
352 Q[nN + m] = 1 / (1 + DD[nN + m]);
353 sum_Q += Q[nN + m];
354 }
355 else Q[nN + m] = DBL_MIN;
356 }
357 nN += N;
358 }
359 for(int i = 0; i < N * N; i++) Q[i] /= sum_Q;
360
361 // Sum t-SNE error
362 double C = .0;
363 for(int n = 0; n < N * N; n++) {
364 C += P[n] * log((P[n] + FLT_MIN) / (Q[n] + FLT_MIN));
365 }
366
367 // Clean up memory
368 free(DD);
369 free(Q);
370 return C;
371}

373// Evaluate t-SNE cost function (approximately)
374static double evaluateError(unsigned int* row_P, unsigned int* col_P, double* val_P, double* Y, int N, int D, double theta)
375{
376
377 // Get estimate of normalization term
378 SPTree* tree = new SPTree(D, Y, N);
379 double* buff = (double*) calloc(D, sizeof(double));
380 double sum_Q = .0;
381 for(int n = 0; n < N; n++) tree->computeNonEdgeForces(n, theta, buff, &sum_Q);
382
383 // Loop over all edges to compute t-SNE error
384 int ind1, ind2;
385 double C = .0, Q;
386 for(int n = 0; n < N; n++) {
387 ind1 = n * D;
388 for(int i = row_P[n]; i < row_P[n + 1]; i++) {
389 Q = .0;
390 ind2 = col_P[i] * D;
391 for(int d = 0; d < D; d++) buff[d] = Y[ind1 + d];
392 for(int d = 0; d < D; d++) buff[d] -= Y[ind2 + d];
393 for(int d = 0; d < D; d++) Q += buff[d] * buff[d];
394 Q = (1.0 / (1.0 + Q)) / sum_Q;
395 C += val_P[i] * log((val_P[i] + FLT_MIN) / (Q + FLT_MIN));
396 }
397 }
398
399 // Clean up memory
400 free(buff);
401 delete tree;
402 return C;
403}

406// Compute input similarities with a fixed perplexity
407static void computeGaussianPerplexity(double* X, int N, int D, double* P, double perplexity) {
408
409 // Compute the squared Euclidean distance matrix
410 double* DD = (double*) malloc(N * N * sizeof(double));
411 if(DD == NULL) { printf("Memory allocation failed!\n"); exit(1); }
412 computeSquaredEuclideanDistance(X, N, D, DD);
413
414 // Compute the Gaussian kernel row by row
415 int nN = 0;
416 for(int n = 0; n < N; n++) {
417
418 // Initialize some variables
419 bool found = false;
420 double beta = 1.0;
421 double min_beta = -DBL_MAX;
422 double max_beta = DBL_MAX;
423 double tol = 1e-5;
424 double sum_P;
425
426 // Iterate until we found a good perplexity
427 int iter = 0;
428 while(!found && iter < 200) {
429
430 // Compute Gaussian kernel row
431 for(int m = 0; m < N; m++) P[nN + m] = exp(-beta * DD[nN + m]);
432 P[nN + n] = DBL_MIN;
433
434 // Compute entropy of current row
435 sum_P = DBL_MIN;
436 for(int m = 0; m < N; m++) sum_P += P[nN + m];
437 double H = 0.0;
438 for(int m = 0; m < N; m++) H += beta * (DD[nN + m] * P[nN + m]);
439 H = (H / sum_P) + log(sum_P);
440
441 // Evaluate whether the entropy is within the tolerance level
442 double Hdiff = H - log(perplexity);
443 if(Hdiff < tol && -Hdiff < tol) {
444 found = true;
445 }
446 else {
447 if(Hdiff > 0) {
448 min_beta = beta;
449 if(max_beta == DBL_MAX || max_beta == -DBL_MAX)
450 beta *= 2.0;
451 else
452 beta = (beta + max_beta) / 2.0;
453 }
454 else {
455 max_beta = beta;
456 if(min_beta == -DBL_MAX || min_beta == DBL_MAX){
457 if (beta < 0) {
458 beta *= 2;
459 } else {
460 beta = beta <= 1.0 ? -0.5 : beta / 2.0;
461 }
462 } else {
463 beta = (beta + min_beta) / 2.0;
464 }
465 }
466 }
467
468 // Update iteration counter
469 iter++;
470 }
471
472 // Row normalize P
473 for(int m = 0; m < N; m++) P[nN + m] /= sum_P;
474 nN += N;
475 }
476
477 // Clean up memory
478 free(DD); DD = NULL;
479}

482// Compute input similarities with a fixed perplexity using ball trees (this function allocates memory another function should free)
483static void computeGaussianPerplexity(double* X, int N, int D, unsigned int** _row_P, unsigned int** _col_P, double** _val_P, double perplexity, int K) {
484
485 if(perplexity > K) printf("Perplexity should be lower than K!\n");
486
487 // Allocate the memory we need
488 *_row_P = (unsigned int*) malloc((N + 1) * sizeof(unsigned int));
489 *_col_P = (unsigned int*) calloc(N * K, sizeof(unsigned int));
490 *_val_P = (double*) calloc(N * K, sizeof(double));
491 if(*_row_P == NULL || *_col_P == NULL || *_val_P == NULL) { printf("Memory allocation failed!\n"); exit(1); }
492 unsigned int* row_P = *_row_P;
493 unsigned int* col_P = *_col_P;
494 double* val_P = *_val_P;
495 double* cur_P = (double*) malloc((N - 1) * sizeof(double));
496 if(cur_P == NULL) { printf("Memory allocation failed!\n"); exit(1); }
497 row_P[0] = 0;
498 for(int n = 0; n < N; n++) row_P[n + 1] = row_P[n] + (unsigned int) K;
499
500 // Build ball tree on data set
501 VpTree<DataPoint, euclidean_distance>* tree = new VpTree<DataPoint, euclidean_distance>();
502 vector<DataPoint> obj_X(N, DataPoint(D, -1, X));
503 for(int n = 0; n < N; n++) obj_X[n] = DataPoint(D, n, X + n * D);
504 tree->create(obj_X);
505
506 // Loop over all points to find nearest neighbors
507 printf("Building tree...\n");
508 vector<DataPoint> indices;
509 vector<double> distances;
510 for(int n = 0; n < N; n++) {
511
512 if(n % 10000 == 0) printf(" - point %d of %d\n", n, N);
513
514 // Find nearest neighbors
515 indices.clear();
516 distances.clear();
517 tree->search(obj_X[n], K + 1, &indices, &distances);
518
519 // Initialize some variables for binary search
520 bool found = false;
521 double beta = 1.0;
522 double min_beta = -DBL_MAX;
523 double max_beta = DBL_MAX;
524 double tol = 1e-5;
525
526 // Iterate until we found a good perplexity
527 int iter = 0; double sum_P;
528 while(!found && iter < 200) {
529
530 // Compute Gaussian kernel row
531 for(int m = 0; m < K; m++) cur_P[m] = exp(-beta * distances[m + 1] * distances[m + 1]);
532
533 // Compute entropy of current row
534 sum_P = DBL_MIN;
535 for(int m = 0; m < K; m++) sum_P += cur_P[m];
536 double H = .0;
537 for(int m = 0; m < K; m++) H += beta * (distances[m + 1] * distances[m + 1] * cur_P[m]);
538 H = (H / sum_P) + log(sum_P);
539
540 // Evaluate whether the entropy is within the tolerance level
541 double Hdiff = H - log(perplexity);
542 if(Hdiff < tol && -Hdiff < tol) {
543 found = true;
544 }
545 else {
546 if(Hdiff > 0) {
547 min_beta = beta;
548 if(max_beta == DBL_MAX || max_beta == -DBL_MAX)
549 beta *= 2.0;
550 else
551 beta = (beta + max_beta) / 2.0;
552 }
553 else {
554 max_beta = beta;
555 if(min_beta == -DBL_MAX || min_beta == DBL_MAX)
556 beta /= 2.0;
557 else
558 beta = (beta + min_beta) / 2.0;
559 }
560 }
561
562 // Update iteration counter
563 iter++;
564 }
565
566 // Row-normalize current row of P and store in matrix
567 for(unsigned int m = 0; m < K; m++) cur_P[m] /= sum_P;
568 for(unsigned int m = 0; m < K; m++) {
569 col_P[row_P[n] + m] = (unsigned int) indices[m + 1].index();
570 val_P[row_P[n] + m] = cur_P[m];
571 }
572 }
573
574 // Clean up memory
575 obj_X.clear();
576 free(cur_P);
577 delete tree;
578}

581// Symmetrizes a sparse matrix
582static void symmetrizeMatrix(unsigned int** _row_P, unsigned int** _col_P, double** _val_P, int N) {
583
584 // Get sparse matrix
585 unsigned int* row_P = *_row_P;
586 unsigned int* col_P = *_col_P;
587 double* val_P = *_val_P;
588
589 // Count number of elements and row counts of symmetric matrix
590 int* row_counts = (int*) calloc(N, sizeof(int));
591 if(row_counts == NULL) { printf("Memory allocation failed!\n"); exit(1); }
592 for(int n = 0; n < N; n++) {
593 for(int i = row_P[n]; i < row_P[n + 1]; i++) {
594
595 // Check whether element (col_P[i], n) is present
596 bool present = false;
597 for(int m = row_P[col_P[i]]; m < row_P[col_P[i] + 1]; m++) {
598 if(col_P[m] == n) present = true;
599 }
600 if(present) row_counts[n]++;
601 else {
602 row_counts[n]++;
603 row_counts[col_P[i]]++;
604 }
605 }
606 }
607 int no_elem = 0;
608 for(int n = 0; n < N; n++) no_elem += row_counts[n];
609
610 // Allocate memory for symmetrized matrix
611 unsigned int* sym_row_P = (unsigned int*) malloc((N + 1) * sizeof(unsigned int));
612 unsigned int* sym_col_P = (unsigned int*) malloc(no_elem * sizeof(unsigned int));
613 double* sym_val_P = (double*) malloc(no_elem * sizeof(double));
614 if(sym_row_P == NULL || sym_col_P == NULL || sym_val_P == NULL) { printf("Memory allocation failed!\n"); exit(1); }
615
616 // Construct new row indices for symmetric matrix
617 sym_row_P[0] = 0;
618 for(int n = 0; n < N; n++) sym_row_P[n + 1] = sym_row_P[n] + (unsigned int) row_counts[n];
619
620 // Fill the result matrix
621 int* offset = (int*) calloc(N, sizeof(int));
622 if(offset == NULL) { printf("Memory allocation failed!\n"); exit(1); }
623 for(int n = 0; n < N; n++) {
624 for(unsigned int i = row_P[n]; i < row_P[n + 1]; i++) { // considering element(n, col_P[i])
625
626 // Check whether element (col_P[i], n) is present
627 bool present = false;
628 for(unsigned int m = row_P[col_P[i]]; m < row_P[col_P[i] + 1]; m++) {
629 if(col_P[m] == n) {
630 present = true;
631 if(n <= col_P[i]) { // make sure we do not add elements twice
632 sym_col_P[sym_row_P[n] + offset[n]] = col_P[i];
633 sym_col_P[sym_row_P[col_P[i]] + offset[col_P[i]]] = n;
634 sym_val_P[sym_row_P[n] + offset[n]] = val_P[i] + val_P[m];
635 sym_val_P[sym_row_P[col_P[i]] + offset[col_P[i]]] = val_P[i] + val_P[m];
636 }
637 }
638 }
639
640 // If (col_P[i], n) is not present, there is no addition involved
641 if(!present) {
642 sym_col_P[sym_row_P[n] + offset[n]] = col_P[i];
643 sym_col_P[sym_row_P[col_P[i]] + offset[col_P[i]]] = n;
644 sym_val_P[sym_row_P[n] + offset[n]] = val_P[i];
645 sym_val_P[sym_row_P[col_P[i]] + offset[col_P[i]]] = val_P[i];
646 }
647
648 // Update offsets
649 if(!present || (present && n <= col_P[i])) {
650 offset[n]++;
651 if(col_P[i] != n) offset[col_P[i]]++;
652 }
653 }
654 }
655
656 // Divide the result by two
657 for(int i = 0; i < no_elem; i++) sym_val_P[i] /= 2.0;
658
659 // Return symmetrized matrices
660 free(*_row_P); *_row_P = sym_row_P;
661 free(*_col_P); *_col_P = sym_col_P;
662 free(*_val_P); *_val_P = sym_val_P;
663
664 // Free up some memery
665 free(offset); offset = NULL;
666 free(row_counts); row_counts = NULL;
667}

669// Compute squared Euclidean distance matrix
670static void computeSquaredEuclideanDistance(double* X, int N, int D, double* DD) {
671 const double* XnD = X;
672 for(int n = 0; n < N; ++n, XnD += D) {
673 const double* XmD = XnD + D;
674 double* curr_elem = &DD[n*N + n];
675 *curr_elem = 0.0;
676 double* curr_elem_sym = curr_elem + N;
677 for(int m = n + 1; m < N; ++m, XmD+=D, curr_elem_sym+=N) {
678 *(++curr_elem) = 0.0;
679 for(int d = 0; d < D; ++d) {
680 *curr_elem += (XnD[d] - XmD[d]) * (XnD[d] - XmD[d]);
681 }
682 *curr_elem_sym = *curr_elem;
683 }
684 }
685}

718static double randn() {
719 double x, y, radius;
720 do {
721 x = 2 * (rand() / ((double) RAND_MAX + 1)) - 1;
722 y = 2 * (rand() / ((double) RAND_MAX + 1)) - 1;
723 radius = (x * x) + (y * y);

725 radius = sqrt(-2 * log(radius) / radius);
726 x *= radius;
